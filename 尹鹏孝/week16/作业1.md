# 设置键值对
SET key value
SET key value EX seconds        # 设置过期时间（秒）
SET key value PX milliseconds  # 设置过期时间（毫秒）
SET key value NX               # 仅当键不存在时设置
SET key value XX               # 仅当键存在时设置

# 获取值
GET key

# 检查键是否存在
EXISTS key [key ...]

# 删除键
DEL key [key ...]

# 设置过期时间
EXPIRE key seconds
PEXPIRE key milliseconds
EXPIREAT key timestamp         # Unix时间戳
PEXPIREAT key milliseconds-timestamp

# 移除过期时间
PERSIST key

# 查看剩余生存时间
TTL key      # 返回秒
PTTL key     # 返回毫秒

# 重命名键
RENAME key newkey
RENAMENX key newkey           # 仅当新键不存在时重命名

# 查找键
KEYS pattern                  # pattern支持 * ? []
SCAN cursor [MATCH pattern] [COUNT count]

# 获取键类型
TYPE key
