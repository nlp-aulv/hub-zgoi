# 对多模态RAG的项目，设计下接口，需要满足：数据管理的接口、多模态检索接口、多模态问答接口
    接口定义定义清楚，传入的参数 + 返回的结果格式

## RAG项目数据管理接口设计：
### 1、列表接口
url:/rga/list
method:"get"
query:{
    pageSize:10,
    page:1,
    sort:"desc updateTime"
}
result:{
    code:200,
    msg:"查找成功",
    data:{
        total:100,
        list:[{
            id:1,
            name:"这是一个rag列表",
            updateTime:"2025-11-12",
            createName:"ypx",
            updateName:"user",
            describe:"string", 
        }]
    }
}
### 2、创建接口
url:/rga/create
method:"create"
data:{
   
    name:10,
    userId:1,
    describe:"请识别图中的歌手",
    media:'阿来阿木-你的万水千山.png',
}
result:{
    code:200,
    msg:"创建成功",
}
### 3、修改接口
url:/rga/update
method:"put"
data:{
    id:2,
    name:10,
    userId:1,
    describe:"请识别图中的歌手",
    media:'周杰伦-稻香.png',
}
result:{
    code:200,
    msg:"更新成功",
}

### 4、删除接口
url:/rga/delete?id=2
method:"delete"
result:{
    code:200,
    msg:"删除成功",
}

### 5、知识问答接口：
url:/rag/search
method:"post",
data:{
    describe:"请识别图中的歌手",
    media:'周杰伦-稻香.png',
}
result:{
    code:200,
    msg:"删除成功",
data:{
   data: {"type": "partial", "content": "这是"},
    data: {"type": "partial", "content": "回答内容"},
    data: {"type": "complete", "content": "完整的回答"}
}
}
